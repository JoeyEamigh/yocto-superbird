# Bridgething u-boot env -- stock Amlogic partition layout (A/B).
#
# Key differences from the legacy MBR-overlay env:
#   - boot_part=boot lets stock u-boot's imgread auto-resolve to
#     boot_a or boot_b based on active_slot.
#   - sys_part (mmcblk0p14 / p15) follows from active_slot and goes
#     into the kernel root= cmdline.
#   - bootargs is built in u-boot from initargs + sys_part + slot
#     suffix; the FAT bootargs.txt mechanism is gone.
#   - Display init still drives the panel from u-boot via logo
#     partition assets (bootup_spotify, burn_mode).
#   - want_boot=burn / want_boot=kernel one-shot iteration story
#     stays intact so the autonomous reset loop still works.

# basic superbird uboot stuff
EnableSelinux=
Irq_check_en=
avb2=
bcb_cmd=
board_revision=4
boot_part=boot
jtag=

# A/B slot. Default _a; bridgething-ab flips it on successful OTA.
active_slot=_a

# Per-slot boot try counters. boot_safety decrements the active
# slot's counter on each kernel-boot attempt and swaps active_slot
# to the other slot when it reaches 0. bridgething-ab boot-confirm
# (a multi-user.target oneshot) resets the active slot's counter
# to 3 once the system is healthy, so a panic loop is what
# eventually triggers the swap, not a slow first boot.
slot_a_try=3
slot_b_try=3

# serial
baudrate=115200
stderr=serial
stdin=serial
stdout=serial

# display: u-boot drives the MIPI-DSI panel via Amlogic's stock
# panel_type path (display_stack=boe, panel_type=lcd_8). Without
# this preboot the backlight stays dark in u-boot and the kernel's
# st7701 driver has nothing to hand off from.
display_bpp=16
display_color_bg=0
display_color_fg=0xffff
display_color_index=16
display_height=800
display_init=1
display_layer=osd0
display_stack=boe
display_width=480
fb_addr=0x1f800000
fb_height=800
fb_width=480
frac_rate_policy=1
osd_reverse=0
outputmode=panel
panel_type=lcd_8
video_reverse=0

# misc
sdcburncfg=aml_sdc_burn.ini
sdc_burning=sdc_burn ${sdcburncfg}
reboot_mode=normal
reboot_mode_android=
recovery_offset=0
recovery_part=recovery
wipe_cache=successful
wipe_data=successful
system_mode=1

# general configuration. uimgaddr is where we drop the WHOLE
# boot.img blob (Android v0 with kernel + dtb in second-stage);
# bootm walks the header from there. 0x8000000 is clear of both
# the secmon reserved region (0x05000000-0x072fffff in the mainline
# meson-g12a.dtsi) and u-boot's relocated stack at 0x16e42000+.
uimgaddr=0x8000000
loadaddr=0x1080000
fdt_addr=0x1000000
lock=10001000
silent=off
upgrade_step=0
verify=no

# always have firstboot after flash; superbird-init clears it.
firstboot=1

# base kernel cmdline. The slot-aware bits (root, slot_suffix) get
# appended in storeargs. ramoops keeps a kernel log across resets.
fs_type=ro rootwait
initargs=ramoops.pstore_en=1 ramoops.record_size=0x8000 ramoops.console_size=0x4000 console=ttyAML0,115200n8 earlycon=meson,0xff803000 no_console_suspend

# boot checks: GPIOA_3 held low -> USB burn mode, GPIOAO_3 -> drop to
# u-boot shell (start_shell). Both are escape hatches.
check_button=if gpio input GPIOA_3; then run do_usb_burning; elif gpio input GPIOAO_3; then run start_shell; fi;
check_temp=boot_cooling

# boot functions. init_display + do_usb_burning both call vout
# output panel,enable so the DSI is hot regardless of which path we
# take. Logo names match the assets baked into the logo partition
# by the superbird-logo recipe.
update=run usb_burning;
init_display=if test ${display_init} = 1; then osd open;osd clear;imgread pic logo bootup_spotify $loadaddr;bmp display $bootup_spotify_offset;bmp scale;vout output ${outputmode};fi;
start_shell=update 1; setenv bootcmd "amlmmc env";
splash_boot=run storeboot;
do_usb_burning=osd open;osd clear;imgread pic logo burn_mode $loadaddr;bmp display $burn_mode_offset;bmp scale;vout output ${outputmode}; update 1000;
usb_burning=update 1000
try_auto_burn=update 700 750;

# A/B slot wiring. sys_part is the Linux-side partition node the
# kernel mounts as root; boot_part / dtb_part are the Amlogic
# partition names we read the boot.img and our dtb from. We
# overlay an MBR onto sector 0 of bootloader.dump that exposes the
# AML system_a/b/settings/data partitions to mainline as
# mmcblk0p1..p4 (mainline can't read the AML MPT directly). u-boot
# still uses AML names via amlmmc.
get_boot_slot=if test ${active_slot} = _a; then setenv sys_part mmcblk0p1; setenv boot_part boot_a; setenv dtb_part dtbo_a; elif test ${active_slot} = _b; then setenv sys_part mmcblk0p2; setenv boot_part boot_b; setenv dtb_part dtbo_b; fi;

# bootargs construction. ${initargs} carries the static console +
# ramoops bits; root=/dev/${sys_part} flips between system_a and
# system_b; androidboot.slot_suffix lets userspace know which slot
# is live (consumed by bridgething's AB module).
storeargs=run get_boot_slot; setenv bootargs ${initargs} root=/dev/${sys_part} ${fs_type} androidboot.slot_suffix=${active_slot} bridgething.firstboot=${firstboot};

# boot_safety: decrement the active slot's try counter; if it hit
# 0 last cycle, swap active_slot to the other slot. setexpr is in
# stock u-boot v1.0-40 so we can do arithmetic; if it ever isn't
# the test-ladder fallback is straightforward to write.
boot_safety=if test "${active_slot}" = "_a"; then if test "${slot_a_try}" -le "0"; then setenv active_slot _b; saveenv; echo "[boot_safety] slot _a try counter exhausted, swapping to _b"; else setexpr slot_a_try ${slot_a_try} - 1; saveenv; fi; elif test "${active_slot}" = "_b"; then if test "${slot_b_try}" -le "0"; then setenv active_slot _a; saveenv; echo "[boot_safety] slot _b try counter exhausted, swapping to _a"; else setexpr slot_b_try ${slot_b_try} - 1; saveenv; fi; fi;

# boot sequence. Default = boot to kernel. To recover into burn
# mode from a running system, `fw_setenv want_boot burn && reboot`
# from userland (libubootenv-bin ships fw_setenv; /etc/fw_env.config
# pins the env partition). bridgething-boot-confirm.service still
# resets the active slot's try counter on every successful kernel
# boot, but does NOT touch want_boot anymore -- userspace owns that
# now. boot_safety still walks the per-slot try counter so a panic
# loop on slot _a auto-swaps to _b after three failed attempts.
want_boot=kernel
preboot=run init_display; run check_button;
bootcmd=if test "${want_boot}" = "kernel"; then run check_temp; run boot_safety; run storeboot; else run do_usb_burning; fi;

# storeboot. Three things to know:
#  - secure-boot efuse on this unit makes `imgread kernel` reject
#    unsigned images, so we use `amlmmc read` (no sigcheck) plus
#    the bootm subcommand chain (also skips the sigcheck the
#    monolithic `bootm` does).
#  - Stock u-boot keeps a stale multi-dtb at ${fdt_addr} from its
#    own preboot, and bootm doesn't pick up the dtb from the
#    boot.img's second-stage slot. We pre-load our dtb from
#    dtbo_${slot} into ${fdt_addr} so bootm fdt finds the right
#    one. dtbo partition is 4MB, our dtb is ~50KB.
#  - We pull the WHOLE 16MB boot_X partition into ${uimgaddr};
#    bootm walks the Android header from there.
storeboot=run storeargs; if amlmmc read ${boot_part} ${uimgaddr} 0 0x1000000; then if amlmmc read ${dtb_part} ${fdt_addr} 0 0x10000; then bootm start ${uimgaddr}; bootm loados; bootm fdt; bootm prep; bootm go; fi; fi; update 1000;
