# u-boot environment for the Spotify Car Thing - mainline u-boot path.
#
# elle's mainline u-boot ships an intentionally policy-free default
# environment (no boot machinery at all). u-boot loads this file as
# uboot.env from the GPT partition named env. It supplies the boot
# policy: extlinux-boot slot A from the GPT partition named boot_a.
#
# The boot partition is referenced by its GPT label, not a hardcoded
# p-number: `part number` resolves boot_a -> its partition index, then
# sysboot reads extlinux.conf from it.
#
# Milestone 1: hardcoded slot A. A/B selection lands with OTA.
stdin=serial
stdout=serial
stderr=serial
fdt_addr_r=0x01000000
kernel_addr_r=0x08080000
scriptaddr=0x08000000
pxefile_addr_r=0x01080000
ramdisk_addr_r=0x13000000
# Image.gz is gzip-compressed; booti decompresses it into this scratch
# buffer. kernel_comp_size must exceed the *decompressed* Image (~44 MiB).
kernel_comp_addr_r=0x0a000000
kernel_comp_size=0x4000000
bootcmd=part number mmc 0 boot_a boot_partnum && sysboot mmc 0:${boot_partnum} any ${scriptaddr} /extlinux/extlinux.conf
